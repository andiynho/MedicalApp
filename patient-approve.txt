	Connection con=DB.getConnection();
	String name = request.getParameter("name-patient");
	String CNP = request.getParameter("CNP-patient");
	String email = request.getParameter("email-patient");
	String user = request.getParameter("username-patient");
	String pass = request.getParameter("password-patient");
	String m_id = request.getParameter("m-id-patient");
	
	if(name.equals("") || CNP.equals("") || email.equals("") || user.equals("") || pass.equals("")){
		JOptionPane.showMessageDialog(null, "Please fill in all the fields!");
		response.sendRedirect("patient-create.jsp");
	}else {
		
		
		PreparedStatement username = con.prepareStatement("INSERT INTO account(username,password,CNP,name,email,account_type,m_personnel_ID) VALUES (?,?,?,?,?,?,?);");
		
		username.setString(1, user);
		username.setString(2, pass);
		username.setString(3, CNP);
		username.setString(4, name);
		username.setString(5, email);
		username.setString(6, "patient");
		username.setString(7, m_id);
		int x = username.executeUpdate();
		if(x > 0)
		{
			JOptionPane.showMessageDialog(null, "Account Created Successfully");
			response.sendRedirect("admin-account.jsp");
		}
		else
		{
			JOptionPane.showMessageDialog(null, "Account Not Created");
			response.sendRedirect("admin-account.jsp");
		}
